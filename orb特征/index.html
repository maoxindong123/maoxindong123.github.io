手写ORB特征检测

2024/2/12 slam slam orb

ORB特征检测

ORB特征检测分：提取关键点、计算描述子、暴力匹配

提取关键点一般会先建立图像金字塔，然后对每一层单独地提取关键点。判断是否为关键点的依据是：选当前点周围的16个点（16个点的图），看是否有连续的N个点的像素值都大于当前点的像素值\(I_p\)一定量\(T\)或者小于，即\((i_1\gt I_p+T\bigcap i_2\gt I_p+T \bigcap ... )\bigcup (i_1\lt I_p-T\bigcap i_2\lt I_p-T \bigcap ... )\)。对于关键点再计算一下方向，方向是在关键点周围计算质心然后计算当前点与质心的连线的角度

然后计算描述子，方法是：在关键点的邻域内随机选256对点，每对点的像素值比较得出0 or 1，得到256维01向量作为描述子

暴力匹配是两两比较汉明距离，然后找个阈值筛一下

正确的线应该都是平行的，opencv检测出来的更多，没有不平行的估计是加了ransac之类的去除误匹配的算法了，所以也许我把阈值放宽然后加ransac也可以更多

vector<int> dy={-3,-3,-2,-1,0,1,2,3,3,3,2,1,0,-1,-2,-3}, dx={0,1,2,3,3,3,2,1,0,-1,-2,-3,-3,-3,-2,-1};

bool cmp(DMatch a,DMatch b)
{
    return a.distance<b.distance;
}

// 算一个点的角度
float calc_angle(Mat im, int x,int y)
{
    //图像块就找以它为中心的往上3往下3的正方形吧，返回的是角度而不是弧度，因为Keypoint里的angle也是角度，范围是0~360
    float m00=0,m01=0,m10=0;
    for(int dy=-3;dy<=3;dy++)
        for(int dx=-3;dx<=3;dx++)
            m00+=im.at<uchar>(y+dy,x+dy),m01+=dy*im.at<uchar>(y+dy,x+dx),m10+=dx*im.at<uchar>(y+dy,x+dx);
    float angle=atan2(m01,m10)*180/PI;
    if(angle<0)angle+=360;
    return angle;
}

// 给一张图，求出整张图的harris响应值
void harris_response(Mat im, Mat &response)
{
    Mat Ix,Iy,xkernel=(Mat_<float>(1,3) << -1, 0, 1),ykernel=(Mat_<float>(3,1) << 1, 0, -1);
    filter2D(im, Ix, CV_32F, xkernel),filter2D(im, Iy, CV_32F, ykernel);
    Mat Ix2=Ix.mul(Ix),Iy2=Iy.mul(Iy),Ixy=Ix.mul(Iy),gaussiankernel= getGaussianKernel(3,1)*getGaussianKernel(3,1).t();
    filter2D(Ix2, Ix2, CV_32F, gaussiankernel),filter2D(Iy2, Iy2, CV_32F, gaussiankernel),filter2D(Ixy, Ixy, CV_32F, gaussiankernel);
    for (int y = 0; y < im.rows; y++)
    {
        for (int x = 0; x < im.cols; x++)
        {
            float det_m = Ix2.at<float>(y,x) * Iy2.at<float>(y,x) - Ixy.at<float>(y,x) * Ixy.at<float>(y,x);
            float trace_m = Ix2.at<float>(y,x) + Iy2.at<float>(y,x);
            response.at<float>(y,x) = det_m - 0.01 * trace_m *trace_m;
        }
    }
}

void build_pyramid(Mat im, vector<Mat> &pyramid)
{
    pyramid.push_back(im);
    for(int i=1;i<L;i++)
    {
        Mat im1=Mat(pyramid[i-1].rows/2,pyramid[i-1].cols/2,CV_8U);
        for(int y=0;y<im1.rows;y++)
            for(int x=0;x<im1.cols;x++)
                im1.at<uchar>(y,x)=pyramid[i-1].at<uchar>(y*2,x*2);
        pyramid.push_back(im1);
    }
}

void extract_keypoints(vector<Mat> py, vector<KeyPoint> &kp)
{
    // 对一张图像先构造图像金字塔，然后每层提取FAST特征点（包括角度和响应值），然后每个点NMS
    // cv::Keypoint类除了邻域大小size、class_id，其他都填上了（坐标pt、方向angle、响应值response、金字塔层数octave）
    // 图像金字塔的不同层找到的点可能属于原图的同一个点，但它的周围环境不同，所以会得到不同的描述子，只要任一描述子匹配，这个点就应该匹配，所以不同层找到的相同点不能去重
    vector<vector<KeyPoint> > all(L,vector<KeyPoint>());
    for(int l=0;l<L;l++)
    {
        Mat response=Mat(py[l].size(), CV_32F);
        harris_response(py[l],response);
        for(int y=B;y<=py[l].rows-B;y++)
        {
            for(int x=B;x<=py[l].cols-B;x++)
            {
                int Ip=py[l].at<uchar>(y,x);
                for(int i=0;i<16;i++)
                {
                    bool gt=true,lt=true;
                    for(int j=0;j<N;j++)
                    {
                        int ip=py[l].at<uchar>(y+dy[(i+j)%16],x+dx[(i+j)%16]);
                        if(gt and ip<Ip+T)
                            gt=false;
                        if(lt and ip>Ip-T)
                            lt=false;
                    }
                    if(gt or lt)
                    {
                        all[l].push_back(KeyPoint(x,y,0, calc_angle(py[l],x,y),response.at<float>(y,x),l));
                        break;
                    }
                }
            }
        }
    }
    // nms
    for(int l=0;l<L;l++)
    {
        Mat mp=Mat::zeros(py[l].size(),CV_8U);
        for(auto p:all[l])mp.at<uchar>(p.pt.y,p.pt.x)=p.response;
        for(auto p:all[l])
        {
            bool dismiss=false;
            for(int y1=-3;y1<=3;y1++)
            {
                for(int x1=-3;x1<=3;x1++)
                {
                    if(y1!=0 and x1!=0 and mp.at<uchar>(p.pt.y+y1,p.pt.x+x1)>=mp.at<uchar>(p.pt.y,p.pt.x))
                        dismiss=true;
                }
            }
            if(!dismiss)kp.push_back(p);
        }
    }
}

void calc_descriptor(vector<Mat> py, vector<KeyPoint> kp, vector<vector<bool> > &dp)
{
    for(int i=0;i<kp.size();i++)
    {
        int x=kp[i].pt.x,y=kp[i].pt.y;
        float angle=kp[i].angle/180*PI;
        vector<bool> tmp(256);
        for(int j=0;j<256;j++)
        {
            Point2f p=Point2f(ORB_pattern[j][0],ORB_pattern[j][1]);
            Point2f q=Point2f(ORB_pattern[j][2],ORB_pattern[j][3]);
            Point2f pp=Point2f(p.x*cos(angle)-p.y*sin(angle),p.x*sin(angle)+p.y*cos(angle));
            Point2f qq=Point2f(q.x*cos(angle)-q.y*sin(angle),q.x*sin(angle)+q.y*cos(angle));
            tmp[j]=(py[kp[i].octave].at<uchar>(y+pp.y,x+pp.x)<py[kp[i].octave].at<uchar >(y+qq.y,x+qq.x));
        }
        dp.push_back(tmp);
    }
}

void bf_match(vector<vector<bool> > dp1, vector<vector<bool> > dp2, vector<DMatch> &match)
{
    // 暴力匹配别写成：遍历一组keypoint，找另一组keypoint谁离当前点最近。这样会出现多个点匹配到另一组的同一个点。改为：n^2地计算汉明距离然后排序
    vector<DMatch> all;
    for(int i=0;i<dp1.size();i++)
    {
        for(int j=0;j<dp2.size();j++)
        {
            int cnt=0;
            for(int k=0;k<256;k++)cnt+=(dp1[i][k]!=dp2[j][k]);
            if(cnt<=T1)all.push_back((DMatch){i,j,(float)cnt});
        }
    }
    sort(all.begin(),all.end(),cmp);
    vector<bool> vis1(dp1.size(),false),vis2(dp2.size(),false);
    for(auto a:all)
    {
        if(a.distance< max(all[0].distance*2,(float)30) and !vis1[a.queryIdx] and !vis2[a.trainIdx])
            match.push_back(a),vis1[a.queryIdx]=true,vis2[a.trainIdx]=true;
    }
}

void myORB(Mat im1, Mat im2, vector<KeyPoint> &kp1, vector<KeyPoint> &kp2, vector<vector<bool> > &dp1, vector<vector<bool> >&dp2, vector<DMatch> &match)
{
    cout<<"图片形状：row"<<im1.rows<<" col"<<im1.cols<<endl;
    // 先建立图像金字塔，然后后面就只传入图像金字塔了
    vector<Mat> py1,py2;
    build_pyramid(im1,py1),build_pyramid(im2,py2);
    extract_keypoints(py1,kp1), extract_keypoints(py2,kp2);
    calc_descriptor(py1,kp1,dp1),calc_descriptor(py2,kp2,dp2);
    bf_match(dp1,dp2,match);
    cout<<"特征点个数："<<kp1.size()<<' '<<kp2.size()<<" match个数："<<match.size()<<endl;
}

int main()
{
    Mat im1= imread("../../pic/sample3.png",0), im2= imread("../../pic/sample4.png",0);
    vector<KeyPoint> kp1,kp2;
    vector<vector<bool> > dp1,dp2; // 与opencv的库函数完全没看齐，opencv是用数据类型是CV_8U也就是uchat的形状是xx*32的Mat存的，但我用xx*256了，并且是二维bool数组
    vector<DMatch> match;
    myORB(im1,im2,kp1,kp2,dp1,dp2,match);


    // 画图
    if(kp1.size()==0 or kp2.size()==0)return 0;
    vector<KeyPoint> kp11,kp22;
    for(auto p:kp1)kp11.push_back((KeyPoint){(float)((int)p.pt.x<<p.octave),(float)((int)p.pt.y<<p.octave),0});
    for(auto p:kp2)kp22.push_back((KeyPoint){(float)((int)p.pt.x<<p.octave),(float)((int)p.pt.y<<p.octave),0});
    Mat out;
    drawMatches(im1, kp11, im2, kp22, match, out);
    imshow("out",out);
    imwrite("out.png",out);
    waitKey(0);
    return 0;
}
